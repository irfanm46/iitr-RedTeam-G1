	HITS Connect Mobile API (Excerpt — OpenAPI-style)																								
																									
	(yaml)																								
	"openapi: 3.0
info:
  title: HITS Connect API
  version: 1.0
servers:
  - url: https://api.hits.ac.in/v1
paths:
  /auth/login:
    post:
      summary: Login
      requestBody:
        content: {application/json: {schema: {type: object,
          properties: {username: {type: string}, password: {type: string}}}}}
      responses: {200: {description: returns JWT token}}
      # NOTE: no rate limiting, no lockout

  /students/{id}:
    get:
      summary: Get student profile by ID
      parameters: [{name: id, in: path, required: true, schema: {type: integer}}]
      # NOTE: token from ANY logged-in student returns ANY id (IDOR)
      responses: {200: {description: full profile incl. phone, aadhaar_masked, grades}}

  /students:
    get:
      summary: List all students
      # NOTE: endpoint present, no auth check enforced (BROKEN)
      responses: {200: {description: array of all students}}

  /fees/{id}:
    get: {summary: Fee details, parameters: [{name: id, in: path}]}

  /attendance/{rollno}:
    get: {summary: Attendance record}"
